class MinStack:

    def __init__(self):
        self.size =0
        self.li = []
        

    def push(self, val: int) -> None:
        self.size += 1        
        self.li.append(val)

    def pop(self) -> None:
        self.size -=1
        self.li.pop()

    def top(self) -> int:
        return self.li[self.size-1]

    def getMin(self) -> int:
        tmp = self.li.copy()
        tmp.sort()
        return tmp[0]
