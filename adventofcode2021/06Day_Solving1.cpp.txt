#include <stdio.h>
char arr[1000000];
char str[500];
int main() {
	scanf("%s", &str);
	int k = 0;
	for (int p = 0; str[p]; p++) {
		if (str[p] != ',') {
			int num = str[p] - '0';
			arr[k++] = num;
		}
	}
	printf("k = %d\n",k);
	int x = 0;
	for (int p = 0; p < 256; p++) {
		for (int i = k; i < k + x; i++) {
			arr[i] = 9;
		}
		k += x;
		x = 0;
		for (int i = 0; i < k; i++) {
			arr[i]--;
			//printf("%d ", arr[i]);
			if (arr[i] == 0) {
				x++;
				arr[i] = 7;
			}
		}
		//printf("\n");
	}
	printf("%d\n", k);
	return 0;
}
