#include <stdio.h>
unsigned long long arr[10];
char str[500];
int k;
int main() {
	scanf("%s", &str);
	unsigned long long k = 0;
	for (int p = 0; str[p]; p++) {
		if (str[p] != ',') {
			int num = str[p] - '0';
			arr[num]++;
			k++;
		}
	}
	for (int p = 0; p < 256; p++) {
		k += arr[0];
		for (int i = 1; i <= 9; i++) {
			arr[i - 1] = arr[i];
		}
		arr[9] = arr[0];
		arr[7] += arr[0];
	}
	printf("%lld\n", k);
	return 0;
}
