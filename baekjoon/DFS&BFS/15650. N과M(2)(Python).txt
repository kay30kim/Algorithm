def dfs(depth, idx):
	if depth == M:
		for i in range(M):
			print(ans[i],end =' ')
		print()
		return
	for i in range(idx,N+1):
		if visited[i] == 0 :
			visited[i] = 1
			ans[depth] = i
			dfs(depth+1,i+1)
			visited[i] = 0

N,M = map(int, input().split())
visited = [0 for _ in range(N+1)]
ans = [0 for _ in range(M)]
dfs(0,1)
