import sys
#dp[n][x] = 끝의 자리 수가 x인 길이가 n인 계단 수의 개수
n = int(sys.stdin.readline())
dp = [[0 for _ in range(10)] for _ in range(n+1)]
for j in range(1,10):
	dp[1][j] = 1

for i in range(2,n+1):
	for j in range(10):
		if j== 0 :
			dp[i][j] = dp[i-1][j+1]
		elif j==9:
			dp[i][j] = dp[i-1][j-1]
		else:
			dp[i][j] = dp[i-1][j-1]+dp[i-1][j+1]
ans =0
for j in range(10):
	ans += dp[n][j]
print(ans%1000000000)
