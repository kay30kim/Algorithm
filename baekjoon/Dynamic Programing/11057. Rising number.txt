import sys
#dp[n][x] = 끝나는 수가 x인 길이가 n인 오르막수의 개수
n = int(sys.stdin.readline())
dp = [[0 for _ in range(10)] for _ in range(n+1)]
for i in range(10):
	dp[1][i] = 1
for i in range(2,n+1):
	for j in range(10):
		for k in range(j+1):
			dp[i][j] += dp[i-1][k]
ans =0
for j in range(10):
	ans +=dp[n][j]
print(ans%10007)
