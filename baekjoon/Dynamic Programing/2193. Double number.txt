'''
d[n] = n자리 이친수의 개수
d[1] = 1
d[2] = 10
d[3] = 101 / 100
d[4] = 1010 / 1001 / 1000
d[5] = 10101 / 10100 / 10010 / 10000 / 10001

d[n][x] = x로 끝나는 n자리수 이친수의 개수
d[1][0] = 0
d[1][1] = 1
d[2][0] = d[1][1] + d[1][0]
d[2][1] = d[1][0]
d[3][0] = d[2][1] + d[2][0]
d[3][1] = d[2][0]
'''
import sys
n = int(sys.stdin.readline())
dp = [[0 for _ in range(2)] for _ in range(n+1)]
dp[1][1] = 1
for i in range(2,n+1):
	dp[i][0] = dp[i-1][1] + dp[i-1][0]
	dp[i][1] = dp[i-1][0]
print(dp[n][0]+dp[n][1])
