import sys

	
if __name__ == "__main__":
	N = int(sys.stdin.readline())
	arr=[]
	for _ in range(N):
		start, finish = map(int, sys.stdin.readline().split())
		arr.append([start,finish])
	arr.sort(key = lambda x : (x[1],x[0]))
	finish_time = 0
	cnt = 0
	for now in arr:
		#print(finish_time, now[0])
		if finish_time  <= now[0]:
			finish_time = now[1]
			cnt+=1
	print(cnt)
